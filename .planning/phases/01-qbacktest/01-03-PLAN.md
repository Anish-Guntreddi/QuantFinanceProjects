# qbacktest/metrics/performance.py — public contract
@dataclass
class MetricsReport:
    gross_sharpe: float
    net_sharpe: float
    cost_bps: float            # total costs / total traded value, in bps (0.0 if no trades)
    sortino: float             # on net returns
    max_drawdown: float        # negative fraction, e.g. -0.12, on net equity
    turnover: float            # annualized: sum(|trade value|) / mean equity / years
    hit_rate: float            # fraction of closed trades with realized_pnl > 0 (nan if none)
    sharpe_ci_low: float       # 95% bootstrap CI on net sharpe
    sharpe_ci_high: float
    total_return: float        # net
    n_trades: int

def compute_metrics(
    equity_curve: pd.Series,        # net equity, DatetimeIndex
    gross_returns: pd.Series,       # per-bar returns before costs
    net_returns: pd.Series,         # per-bar returns after costs
    trade_pnls: list[float],        # realized PnL per closed trade
    total_traded_value: float,      # sum of |qty * fill_price| across fills
    total_costs: float,             # commission + slippage cost total
    periods_per_year: int = 252,
) -> MetricsReport

def bootstrap_sharpe_ci(returns: np.ndarray, n_resamples=1000, confidence_level=0.95, rng=42) -> tuple[float, float]
